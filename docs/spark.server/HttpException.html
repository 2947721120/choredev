HttpException class 

class HttpException implements Exception {
 final String message;
 final Uri uri;

 const HttpException(String this.message, {Uri this.uri});

 String toString() {
   var b = new StringBuffer();
   b.write('HttpException: ');
   b.write(message);
   if (uri != null) {
     b.write(', uri = $uri');
   }
   return b.toString();
 }
}
